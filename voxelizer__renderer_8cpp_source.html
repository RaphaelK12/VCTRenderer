574 void VoxelizerRenderer::UpdateProjectionMatrices(const BoundingBox &sceneBox)
575 {
576  auto axisSize = sceneBox.Extent() * 2.0f;
577  auto &center = sceneBox.Center();
578  volumeGridSize = glm::max(axisSize.x, glm::max(axisSize.y, axisSize.z));
579  voxelSize = volumeGridSize / volumeDimension;
580  auto halfSize = volumeGridSize / 2.0f;
581  // projection matrices
582  auto projection = glm::ortho(-halfSize, halfSize, -halfSize, halfSize, 0.0f,
583  volumeGridSize);
584  // view matrices
585  viewProjectionMatrix[0] = lookAt(center + glm::vec3(halfSize, 0.0f, 0.0f),
586  center, glm::vec3(0.0f, 1.0f, 0.0f));
587  viewProjectionMatrix[1] = lookAt(center + glm::vec3(0.0f, halfSize, 0.0f),
588  center, glm::vec3(0.0f, 0.0f, -1.0f));
589  viewProjectionMatrix[2] = lookAt(center + glm::vec3(0.0f, 0.0f, halfSize),
590  center, glm::vec3(0.0f, 1.0f, 0.0f));
591  int i = 0;
592 
593  for (auto &matrix : viewProjectionMatrix)
594  {
595  matrix = projection * matrix;
596  viewProjectionMatrixI[i++] = inverse(matrix);
597  }
598 }
