215 void GIDeferredRenderer::SetLightPassUniforms() const
216 {
217  static auto &camera = Camera::Active();
218  static auto &scene = Scene::Active();
219  auto &prog = CurrentProgram<LightingProgram>();;
220  prog.cameraPosition.Set(camera->Position());
221  prog.inverseProjectionView.Set(camera->InverseViewMatrix() *
222  camera->InverseProjectionMatrix());
223 
224  for (int i = 0; i < bufferTextures.size(); i++)
225  {
226  bufferTextures[i].Active(i);
227  bufferTextures[i].Bind(oglplus::TextureTarget::_2D);
228  }
229 
230  // uniform arrays of lights
231  auto &uDirectionals = prog.directionalLight;
232  auto &uPoints = prog.pointLight;
233  auto &uSpots = prog.spotLight;
234  auto &lights = scene->lights;
235  // index of directional-point-spot lights
236  auto typeIndex = glm::uvec3(0);
237  // pass number of lights per type
238  prog.lightTypeCount[0].Set(static_cast<const unsigned int>
239  (Light::Directionals().size()));
240  prog.lightTypeCount[1].Set(static_cast<const unsigned int>
241  (Light::Points().size()));
242  prog.lightTypeCount[2].Set(static_cast<const unsigned int>
243  (Light::Spots().size()));
244 
245  for (int i = 0; i < lights.size(); ++i)
246  {
247  auto &light = lights[i];
248  auto &factor = light->Intensities();
249  auto shadowingMethod = light->mode[0].to_ulong();
250  shadowingMethod = shadowingMethod == 2 && sampleVoxelShadowVolume
251  ? 3 : shadowingMethod;
252  // current light uniform
253  auto &uLight = light->Type() == Light::Directional
254  ? uDirectionals[typeIndex.x++]
255  : light->Type() == Light::Point
256  ? uPoints[typeIndex.y++]
257  : uSpots[typeIndex.z++];
258  // shared uniforms between types
259  uLight.shadowingMethod.Set(shadowingMethod);
260  uLight.ambient.Set(light->Ambient() * factor.x);
261  uLight.diffuse.Set(light->Diffuse() * factor.y);
262  uLight.specular.Set(light->Specular() * factor.z);
263 
264  if (light->Type() == Light::Spot || light->Type() == Light::Point)
265  {
266  uLight.position.Set(light->Position());
267  }
268 
269  if (light->Type() == Light::Spot || light->Type() == Light::Directional)
270  {
271  uLight.direction.Set(light->Direction());
272  }
273 
274  if (light->Type() == Light::Spot || light->Type() == Light::Point)
275  {
276  uLight.attenuation.constant.Set(light->attenuation.Constant());
277  uLight.attenuation.linear.Set(light->attenuation.Linear());
278  uLight.attenuation.quadratic.Set(light->attenuation.Quadratic());
279  }
280 
281  if(light->Type() == Light::Spot)
282  {
283  uLight.angleInnerCone.Set(cos(light->AngleInnerCone()));
284  uLight.angleOuterCone.Set(cos(light->AngleOuterCone()));
285  }
286  }
287

289  static auto &shadowing = *static_cast<ShadowMapRenderer *>
290  (AssetsManager::Instance()
291  ->renderers["Shadowmapping"].get());
292 
293  if(shadowing.Caster() != nullptr)
294  {
295  prog.lightViewProjection.Set(shadowing.LightSpaceMatrix());
296  shadowing.BindReading(5);
297  prog.exponents.Set(shadowing.Exponents());
298  prog.lightBleedingReduction.Set(shadowing.LightBleedingReduction());
299  }
300

302  ->renderers["Voxelizer"].get());
303  prog.volumeDimension.Set(voxel.VolumeDimension());
304  prog.voxelScale.Set(1.0f / voxel.VolumeGridSize());
305  prog.worldMinPoint.Set(scene->rootNode->boundaries.MinPoint());
306  prog.worldMaxPoint.Set(scene->rootNode->boundaries.MaxPoint());
307

312  }

320  mips[0].Active(8 + i);
321  mips[0].Bind(oglplus::TextureTarget::_3D);
322  }
323 
324  // global illum setup

326  prog.bounceStrength.Set(globalIlluminationStrength);

380  geometryBuffer.AttachColorTexture(FramebufferTarget::Draw, 2, bufferTextures[2],
381  0);

383  gl.Bound(TextureTarget::_2D, bufferTextures[3])
384  .Image2D(0, PixelDataInternalFormat::RGB8, windowWidth, windowHeight,
385  0, PixelDataFormat::RGB, PixelDataType::UnsignedByte, nullptr)
386  .MinFilter(TextureMinFilter::Nearest)
387  .MagFilter(TextureMagFilter::Nearest);
388  geometryBuffer.AttachColorTexture(FramebufferTarget::Draw, 3, bufferTextures[3],
389  0);

391  gl.Bound(TextureTarget::_2D, bufferTextures[4])

395  .MinFilter(TextureMinFilter::Nearest)
396  .MagFilter(TextureMagFilter::Nearest);
397  geometryBuffer.AttachTexture(FramebufferTarget::Draw,

404  FramebufferColorAttachment::_1,
405  FramebufferColorAttachment::_2,

407  };
