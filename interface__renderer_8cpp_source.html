295 void InterfaceRenderer::CreateDeviceObjects()
296 {
297  // Backup GL state
298  GLint last_texture, last_array_buffer, last_vertex_array;
299  glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
300  glGetIntegerv(GL_ARRAY_BUFFER_BINDING, &last_array_buffer);
301  glGetIntegerv(GL_VERTEX_ARRAY_BINDING, &last_vertex_array);
302  auto vertex_shader =
303  "#version 330\n"
304  "uniform mat4 ProjMtx;\n"
305  "in vec2 Position;\n"
306  "in vec2 UV;\n"
307  "in vec4 Color;\n"
308  "out vec2 Frag_UV;\n"
309  "out vec4 Frag_Color;\n"
310  "void main()\n"
311  "{\n"
312  " Frag_UV = UV;\n"
313  " Frag_Color = Color;\n"
314  " gl_Position = ProjMtx * vec4(Position.xy,0,1);\n"
315  "}\n";
316  auto fragment_shader =
317  "#version 330\n"
318  "uniform sampler2D Texture;\n"
319  "in vec2 Frag_UV;\n"
320  "in vec4 Frag_Color;\n"
321  "out vec4 Out_Color;\n"
322  "void main()\n"
323  "{\n"
324  " Out_Color = Frag_Color * texture( Texture, Frag_UV.st);\n"
325  "}\n";
326  renderer->shaderHandle = glCreateProgram();
327  renderer->vertHandle = glCreateShader(GL_VERTEX_SHADER);
328  renderer->fragHandle = glCreateShader(GL_FRAGMENT_SHADER);
329  glShaderSource(renderer->vertHandle, 1, &vertex_shader, 0);
330  glShaderSource(renderer->fragHandle, 1, &fragment_shader, 0);
331  glCompileShader(renderer->vertHandle);
332  glCompileShader(renderer->fragHandle);
333  glAttachShader(renderer->shaderHandle, renderer->vertHandle);
334  glAttachShader(renderer->shaderHandle, renderer->fragHandle);
335  glLinkProgram(renderer->shaderHandle);
336  renderer->attribLocationTex = glGetUniformLocation
337  (renderer->shaderHandle, "Texture");
338  renderer->attribLocationProjMtx = glGetUniformLocation
339  (renderer->shaderHandle, "ProjMtx");
340  renderer->attribLocationPosition = glGetAttribLocation
341  (renderer->shaderHandle, "Position");
342  renderer->attribLocationUV = glGetAttribLocation
343  (renderer->shaderHandle, "UV");
344  renderer->attribLocationColor = glGetAttribLocation
345  (renderer->shaderHandle, "Color");
346  glGenBuffers(1, &renderer->vboHandle);
347  glGenBuffers(1, &renderer->elementsHandle);
348  glGenVertexArrays(1, &renderer->vaoHandle);
349  glBindVertexArray(renderer->vaoHandle);
350  glBindBuffer(GL_ARRAY_BUFFER, renderer->vboHandle);
351  glEnableVertexAttribArray(renderer->attribLocationPosition);
352  glEnableVertexAttribArray(renderer->attribLocationUV);
353  glEnableVertexAttribArray(renderer->attribLocationColor);
354 #define OFFSETOF(TYPE, ELEMENT) ((size_t)&(((TYPE *)0)->ELEMENT))
355  glVertexAttribPointer(renderer->attribLocationPosition, 2, GL_FLOAT,
356  GL_FALSE, sizeof(ImDrawVert), (GLvoid *)
357  OFFSETOF(ImDrawVert, pos));
358  glVertexAttribPointer(renderer->attribLocationUV, 2, GL_FLOAT,
359  GL_FALSE, sizeof(ImDrawVert), (GLvoid *)
360  OFFSETOF(ImDrawVert, uv));
361  glVertexAttribPointer(renderer->attribLocationColor, 4, GL_UNSIGNED_BYTE,
362  GL_TRUE, sizeof(ImDrawVert), (GLvoid *)
363  OFFSETOF(ImDrawVert, col));
364 #undef OFFSETOF
365  CreateFontsTexture();
366  // Restore modified GL state
367  glBindTexture(GL_TEXTURE_2D, last_texture);
368  glBindBuffer(GL_ARRAY_BUFFER, last_array_buffer);
369  glBindVertexArray(last_vertex_array);
